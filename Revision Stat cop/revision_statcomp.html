sorted_data <- arrange(df, Month)
sorted_data %>%
  ggplot(aes(x = factor(Month), y = Accident, colour = factor(Month))) +
  geom_boxplot() +
  xlab("Month") +
  theme_tufte() +
  scale_x_discrete(breaks = c(1, 3, 5, 7, 9, 11), labels = c(
    "Jan", "Mar", "May", "Jul", "Sep",
    "Nov"
  )) +
  labs(colour = "Months") +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))
